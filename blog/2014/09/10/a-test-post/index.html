import numpy as np
import matplotlib.pyplot as plt

v = 3.0
n = 2
N = 100
x = np.linspace(-v,v,N)
y = np.linspace(-v,v,N)
X,Y = np.meshgrid(x,y)
R = np.power(np.power(X,2) + np.power(Y,2),0.5)
Z = np.multiply(np.sin(R), np.multiply(np.exp(-R),np.power(R,n)))

f, ax = plt.subplots(1,1)

ax.pcolor(X,Y,Z,cmap='hot')
ax.set_xlim(-v,v)
ax.set_ylim(-v,v)
plt.show()
